8 BIT DECIMAL MULTIPLICATION OF TWO NUMBERS:
cr equ 0dh
lf equ 0ah
data segment
table db '0123456789'
n1 db 04h
n2 db 04h
msg db ' The Result is '
asres db 2 dup(?)
db cr,lf,'$'
data ends
code segment
assume cs:code,ds:data
start: mov ax,data
mov ds,ax
mov al,n1
mul n2
aam
lea bx,table
lea si,asres
inc si
and al,00FH
xlat
mov [si],al
dec si
mov al,ah
and al,00FH
xlat
mov [si],al
mov ah,09H
lea dx,msg
int 21h
quit: mov al,0
mov ah,04ch
int 21h
code ends
end start
